
OPEN () {

	MIRO SI ES PRODUCTOR O CONSUMIDOR EN FUNCION DE LA OPCION DE APERTURA
	SI ES CONSUMIDOR
		BLOQUEO
			INCREMENTO EL NUMERO DE CONSUMIDORES
			ESPERAR A QUR HAYA ALGUN PRODUCTOR
		LIBERO
	SI ES PRODUCTOR
		BLOQUEO
			INCREMENTO EL NUMERO DE PRODUCTORES
			ESPERAR A QUR HAYA ALGUN CONSUMIDOR
		LIBERO
	FIN SI

	ABRIR COMUNICACION
	DEVOLVER DESCRIPTOR DE /proc/modFIFO
}

RELEASE () {

	MIRO SI ES PRODUCTOR O CONSUMIDOR EN FUNCION DE LA OPCION DE APERTURA
	SI ES CONSUMIDOR
		BLOQUEO
			REDUZCO EL NUMERO DE CONSUMIDORES
			SI NO HAY CONSUMIDORES NI PRODUCTORES VACIO EL BUFFER
		LIBERO
	SI ES PRODUCTOR
		BLOQUEO
			REDUZCO EL NUMERO DE PRODUCTORES
			SI NO HAY CONSUMIDORES NI PRODUCTORES VACIO EL BUFFER
		LIBERO
	FIN SI

}

READ () {

	MIRO SI PUEDO LEER LOS BYTES REQUERIDOS

	BLOQUEO MUTEX

		ESPERO A QUE HAYA EN EL BUFFER ALGO QUE LEER Y PRODUCTORES
		COMPROBAR FIN DE COMUNICACION Y MANDAR ERROR EN TAL CASO
		EXTRAIGO DATOS
		DESPIERTO A UN POSIBLE PRODUCTOR

	LIBERO MUTEX

	COPIO DATOS AL USUARIO

	DEVUELVO LOS BYTES LEIDOS
}

WRITE () {

	MIRO SI PUEDO ESCRIBIR LOS BYTES REQUERIDOS
	COPIA LOS DATOS DEL USUARIO

	BLOQUEO MUTEX

		ESPERO A QUE HAYA HUECO EN EL BUFFER Y CONSUIMDORES
		COMPROBAR FIN DE COMUNICACION Y MANDAR ERROR EN TAL CASO
		INSERTO DATOS
		DESPIERTO A UN POSIBLE CONSUMIDOR

	LIBERO MUTEX

	DEVUELVO LOS BYTES ESCRITOS
}
